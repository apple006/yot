安装

Mac(64bit) & Linux(64bit)

>>> ./configure
>>> cd src
>>> make

运行 “Hello World”

module hello_world
import Text;

func main()
    |
        const str = "hello world";
    |
    Text.WriteString(str);
    Text.WriteLn
end;
>>> ./yot -o hello_world.yot
>>> ./hello_world
Hello World
